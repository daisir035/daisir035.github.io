...


docker run [可选参数] image     #参数说明

$docker run -d --name nginx01 -p 3344:80 nginx

--name="Name"      容器名字 tomcat01 tomcat02,用来区分容器

-d                 后台方式运行

-it                使用交互方式运行,进入容器查看内容

  

-p                 指定容器的端口   -p 8080:8080

    -p ip:         主机端口:容器端口

    -p             主机端口:容器端口(常用)

    -p             容器端口

    容器端口

-p          随机指定端口

...

docker rm 容器id #正在运行删不了

docker rm -f $(docker ps -aq) # 删除所有容器

docker ps -a -q|xargs docker rm #删除所有的容器

启动和停止容器
...


docker start #启动容器

docker restart #重启容器

docker stop #停止当前正在运行的容器

docker kill #强制停止

  
  

daisir@daisir：~$docker run -d centos # -d 后台运行

daisir@daisir：~$docker run -itd centos # -i keep STDIN open even if not attached
